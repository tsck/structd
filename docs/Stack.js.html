let map = new WeakMap();

/** Class representing a stack */
export class Stack {
  /**
   * Create a stack.
   */
  constructor() {
    map.set(this, []);
  }

  /**
   * Get a string representation of the stack.
   * @returns {string} The string representation of a stack.
   */
  toString() {
    let stack = map.get(this);
    return stack.toString();
  }

  /**
   * Gets and removes the last element in the stack.
   * @returns {*} The last element in the stack.
   */
  pop() {
    let stack = map.get(this);
    return stack.pop();
  }

  /**
   * Pushes and element onto the stack.
   * @param {*} element - Element to be added to the stack.
   */
  push(element) {
    let stack = map.get(this);
    stack.push(element);
  }

  /**
   * Gets the size of the stack.
   * @returns {number} The size of the stack.
   */
  size() {
    let stack = map.get(this);
    return stack.length;
  }

  /**
   * Checks whether or not the stack is empty.
   * @returns {boolean} True if the stack's empty, false if it's not.
   */
  isEmpty() {
    let stack = map.get(this);
    return stack.length === 0;
  }

  /**
   * Removes all elements from the stack.
   */
  clear() {
    let stack = map.get(this);
    stack.splice(0, stack.length);
  }
}
